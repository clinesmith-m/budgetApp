GET ALL SPENDING CATEGORIES:
INTERFACE -> GCAT
BACKEND   -> <NUM CATEGORIES (4 bytes)>
for each category...
    BACKEND -> <CATEGORY NAME LEN (4bytes)>
    BACKEND -> <CATEGORY NAME>
    BACKEND -> <AMOUNT BUDGETED (double)>
    BACKEND -> <AMOUNT UNSPENT (double)>
INTERFACE -> T

GET SINGLE SPENDING CATEGORY:
INTERFACE   -> GCBN       // Get category by name
INTERFACE   -> <CATEGORY NAME>
BACKEND     -> <AMOUNT BUDGETED (DOUBLE)>
BACKEND     -> <AMOUNT SPENT (DOUBLE)>
INTERFACE   -> T

MODIFY SPENDING CATEGORY:
INTERFACE   -> MCAT
INTERFACE   -> <CATEGORY NAME>
INTERFACE   -> <AMOUNT BUDGETED (DOUBLE)>
BACKEND     -> T

ADD SPENDING CATEGORY:
INTERFACE   -> ACAT
INTERFACE   -> <CATEGORY NAME>
INTERFACE   -> <AMOUNT BUDGETED (DOUBLE)>
BACKEND     -> T

ADD MONTHLY EXPENSE:
INTERFACE   -> AMNE
INTERFACE   -> <MEMO>
INTERFACE   -> <AMOUNT (DOUBLE)>
INTERFACE   -> <NUM MONTHS (SHORT)>
BACKEND     -> T

CANCEL MONTHLY EXPENSE:
INTERFACE   -> CMNE
INTERFACE   -> <MEMO>
BACKEND     -> T

GET MONTHLY EXPENSES:
INTERFACE   -> GMNE
BACKEND     -> <NUM MONTHLY EXPS (INT)>
For each expense...
    BACKEND     -> <MEMO>
    BACKEND     -> <AMOUNT (DOUBLE)>
INTERFACE   -> T

ADD MONTHLY INCOME:
INTERFACE   -> AMNI
INTERFACE   -> <MEMO>
INTERFACE   -> <AMOUNT (DOUBLE)>
BACKEND     -> T

CANCEL MONTHLY INCOME:
INTERFACE   -> CMNI
INTERFACE   -> <MEMO>
BACKEND     -> T

GET MONTHLY INCOME STREAMS:
INTERFACE   -> GMNI
BACKEND     -> <NUM MONTHLY INCS (INT)>
For each income stream...
    BACKEND     -> <MEMO>
    BACKEND     -> <AMOUNT (DOUBLE)>
INTERFACE   -> T

LOG ONE TIME EXPENDITURE:
INTERFACE   -> LOTE
INTERFACE   -> <CATEGORY>
INTERFACE   -> <MEMO>
INTERFACE   -> <AMOUNT (DOUBLE)>
BACKEND     -> T

LOG ONE TIME INCOME:
INTERFACE   -> LOTI
INTERFACE   -> <MEMO>
INTERFACE   -> <AMOUNT (DOUBLE)>
BACKEND     -> T

GET PREVIOUS MONTH'S LOGS:
INTERFACE   -> GPML
BACKEND     -> <MONTH'S INCOME (DOUBLE)>
BACKEND     -> <MONTH'S EXPENSES (DOUBLE)>
INTERFACE   -> T

GET CURRENT YEAR'S LOGS:
INTERFACE   -> GCYL
BACKEND     -> <YEAR'S INCOME (DOUBLE)>
BACKEND     -> <YEAR'S EXPENSES (DOUBLE)>
INTERFACE   -> T

GET ALL TIME LOGS:
INTERFACE   -> GATL
BACKEND     -> <ALL TIME INCOME (DOUBLE)>
BACKEND     -> <ALL TIME EXPENSES (DOUBLE)>
INTERFACE   -> T
